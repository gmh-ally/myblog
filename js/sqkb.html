# 简单实现

判断属性是否是对象，是的话，需要递归

function cloneShallow(source) {
  var target = {}
  for (var key in source) {
    // JavaScript中Object对象原型上的hasOwnProperty()用来判断一个属性是定义在对象本身而不是继承自原型链。
    if (Object.prototype.hasOwnProperty.call(source, key)) {
      target[key] = source[key]
    }
  }
  return target
}

function cloneDeep(source) {
  var target = {}
  for (var key in source) {
    if (Object.prototype.hasOwnProperty.call(source, key)) {
      console.log(typeof source[key])
      if (typeof source[key] === 'object') {
        target[key] = cloneDeep(source[key]) // 注意这里
      } else {
        target[key] = source[key]
      }
    }
  }
  return target
}

深拷贝--改进版

// 木易杨
//把判断类型的函数提取出来，包含对数组 和  对象的判断
function isObject(obj) {
  return typeof obj === 'object' && obj != null
}

// 木易杨
function cloneDeep2(source) {
  if (!isObject(source)) return source // 非对象返回自身

  var target = Array.isArray(source) ? [] : {}
  for (var key in source) {
    if (Object.prototype.hasOwnProperty.call(source, key)) {
      if (isObject(source[key])) {
        target[key] = cloneDeep2(source[key]) // 注意这里
      } else {
        target[key] = source[key]
      }
    }
  }
  return target
}

# 考虑各种数据类型的深浅拷贝

  1. JSON.stringify 会忽略什么？ symbol, undefined, function, 循环引用也无法处理

Map, Set, RegExp, Date, ArrayBuffer 和其他内置类型在进行序列化时也会丢失。

let a = {}
let b = new Set()
b.add(11)
a.test = b
let copy = JSON.parse(JSON.stringify(a))

console.log(a, copy) // 对比 a 和 copy 的值, 发现 Set已丢失。

  1. 如果有循环引用, 可以用 JSON.stringify 来处理吗？

  2. 确定是 stringify 会报错， 而不是 parse 会报错吗？

是 stringify 会报错, JSON.stringify 序列化的时候， 如果有循环引用 则会去序列化这个循环引用里的对象， 一直循环往复下去， 如果不报错， 会导致堆栈过大，出现性能问题

  1. 代码实现一个深拷贝
function deepClone(obj, hash = new WeakMap()) {
  if (obj === null) {
    return null
  }

  if (obj instanceof Date) {
    return new Date(obj)
  }

  if (obj instanceof RegExp) {
    return new RegExp(obj)
  }

  if (typeof obj !== 'object') {
    // 不是对象  那就是基本类型了  可以作为递归的终止条件
    return obj
  }

  if (hash.has(obj)) {
    // 处理循环引用的问题  用一个weakmap 做一个弱引用
    return hash.get(obj)
  }

  const resObj = Array.isArray(obj) ? [] : {}

  hash.set(obj, resObj)

  Reflect.ownKeys(obj).forEach((key) => {
    resObj[key] = deepClone(obj[key], hash)
  })

  return resObj
}

# 平时都如何判断对象的类型， 分别适合哪些场景

  • typeof
  • instanceof

  1. instanceof 的内部机制是通过判断对象的原型链中是不是能找到类型的 prototype

使用 instanceof 判断一个对象是否为数组，instanceof 会判断这个对象的原型链上是否会找到对应的 Array 的原型，找到返回 true，否则返回 false。

# 实现一个 instanceof

  • 考察原型链的知识
  • 向前循环迭代
  • 关键就在于 left.proto === right.prototype
function instanceOf(left, right) {
  if (typeof left !== 'object' || left === null) {
    return false
  }

  while (true) {
    if (left === null) {
      return false
    }

    if (left.__proto__ === right.prototype) {
      return true
    }

    left = left.__proto__
  }
}
